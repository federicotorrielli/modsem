Query su GraphDB

Attenzione: i namespace e il nome dei named graphs dipendono dai parametri forniti al sistema alla creazione del repository e dei grafi.

(usare i file di esempio)


#Esempi di INSERT e INSERT in un grafo

PREFIX cinema: <http://www.modsem.org/cinema2#>

INSERT DATA 

{ cinema:StarWars7 cinema:haPersonaggio cinema:PaolinoPaperino}


INSERT DATA 

	{ GRAPH <http://modsem.org/esempi/cinemaWD> 

    	{cinema:StarWars7 cinema:haPersonaggio cinema:PaolinoPaperino}

    }


#Esempio di verifica che esistano delle triple che soddisfano un determinato pattern (anche su un grafo specifico)

ASK  {  

	<http://www.modsem.org/cinema2#StarWars7>

    <http://www.modsem.org/cinema2#haPersonaggio>

    <http://www.modsem.org/cinema2#HanSolo> . 

}


ASK  {  

    

    GRAPH <http://modsem.org/esempi/cinemaWD> {

    

	<http://www.modsem.org/cinema2#StarWars7>

    <http://www.modsem.org/cinema2#haPersonaggio>

    <http://www.modsem.org/cinema2#HanSolo> . 

    }

}


#Esempi di cancellazione di triple

DELETE DATA

{

	cinema:StarWars7 cinema:haPersonaggio cinema:PaolinoPaperino 

}

# QUERY SU INFERENZE

# non estrae l'appartenenza inferita del soggetto alle classi (perché in GraphDB non sono nel grafo <http://modsem.org/esempi/cinemaWD> ma in quello "implicit")

SELECT ?c

FROM <http://modsem.org/esempi/cinemaWD>

WHERE {

	cinema2:StevenSpielberg a ?c .

}


# estrae le inferenze perché non specifica il grafo

SELECT ?c

WHERE {

	cinema2:StevenSpielberg a ?c .

}
