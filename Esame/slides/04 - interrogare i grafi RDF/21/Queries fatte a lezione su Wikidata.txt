#TUTTI I CANTANTI

SELECT ?professione ?professioneLabel WHERE {

  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],it". }

  ?professione wdt:P106 wd:Q177220.

}

LIMIT 100


#CANTANTI ITALIANI

SELECT DISTINCT ?person ?personLabel

WHERE {

  ?person wdt:P106 wd:Q177220.

  ?person wdt:P27 wd:Q38. 

  SERVICE wikibase:label { bd:serviceParam wikibase:language "fr, ar ,en." }

}

#CITTA GIAPPONESI CON POPOLAZIONE

SELECT ?Giappone ?GiapponeLabel ?popolazione WHERE {

  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }

  ?Giappone wdt:P17 wd:Q17.

  ?Giappone wdt:P31 wd:Q515.

  OPTIONAL { ?Giappone wdt:P1082 ?popolazione. }


}

LIMIT 100


#CITTA GIAPPONESI SU MAPPA

#defaultView:Map

SELECT * WHERE {

  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }

  ?Giappone wdt:P17 wd:Q17.

  ?Giappone wdt:P31 wd:Q515.

  ?Giappone wdt:P625 ?geo .

}

LIMIT 100


#Cantanti per nazione

#su bubble chart

#defaultView:BubbleChart

SELECT ?nazioneLabel (COUNT (?professione) AS ?count)  

WHERE {

  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],it". }

  ?professione wdt:P106 wd:Q177220.

  ?professione wdt:P27 ?nazione .

}

GROUP BY ?nazioneLabel

LIMIT 1000


#Cantanti per nazione

#tabella

SELECT ?nazioneLabel (COUNT (?professione) AS ?count)  

WHERE {

  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],it". }

  ?professione wdt:P106 wd:Q177220.

  ?professione wdt:P27 ?nazione .

}

GROUP BY ?nazioneLabel

ORDER BY DESC (?count)
